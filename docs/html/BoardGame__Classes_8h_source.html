270template <typename T>
271class GameManager {
272 Board<T>* boardPtr;
273 Player<T>* players[2];
274 UI<T>* ui;
275
276public:
280 GameManager(Board<T>* b, Player<T>* p[2], UI<T>* u)
281 : boardPtr(b), ui(u) {
282 players[0] = p[0];
283 players[1] = p[1];
284 players[0]->set_board_ptr(b);
285 players[1]->set_board_ptr(b);
286 }
287
291 void run() {
292 ui->display_board_matrix(boardPtr->get_board_matrix());
293 Player<T>* currentPlayer = players[0];
294
295 while (true) {
296 for (int i : {0, 1}) {
297 currentPlayer = players[i];
298 Move<T>* move = ui->get_move(currentPlayer);
299
300 while (!boardPtr->update_board(move))
301 move = ui->get_move(currentPlayer);
302
303 ui->display_board_matrix(boardPtr->get_board_matrix());
304
305 if (boardPtr->is_win(currentPlayer)) {
306 ui->display_message(currentPlayer->get_name() + " wins!");
307 return;
308 }
309 if (boardPtr->is_lose(currentPlayer)) {
310 ui->display_message(players[1 - i]->get_name() + " wins!");
311 return;
312 }
313 if (boardPtr->is_draw(currentPlayer)) {
314 ui->display_message("Draw!");
315 return;
316 }
317 }
318 }
319 }
320};
