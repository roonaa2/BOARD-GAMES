54bool Ultimate_Board::is_win(Player<char>* player) {
55 const char sym = player->get_symbol();
56
57 auto all_equal = [&](char a, char b, char c) {
58 return a == b && b == c && a != blank_symbol;
59 };
60
61 // Lambda to check if a 3x3 board is won by symbol s to avoid repeating the code
62
63 auto has_won = [&](char b[3][3], const char& s) {
64 // Rows & columns
65 for (int i = 0; i < 3; ++i) {
66 if (all_equal(b[i][0], b[i][1], b[i][2]) && b[i][0] == s) return true;
67 if (all_equal(b[0][i], b[1][i], b[2][i]) && b[0][i] == s) return true;
68 }
69 //check Diagonals
70 if (all_equal(b[0][0], b[1][1], b[2][2]) && b[0][0] == s) return true;
71 if (all_equal(b[0][2], b[1][1], b[2][0]) && b[0][2] == s) return true;
72 return false;
73 };
74
75
76 // Check each sub-board and if it is won, mark it in subboard
77
78 for (int sr = 0; sr < 3; ++sr) {
79 for (int sc = 0; sc < 3; ++sc) {
80
81 //if Already won - skip it
82 if (subboard[sr][sc] != blank_symbol)
83 continue;
84
85 // Extract the 3×3 sub-board
86 char small[3][3];
87 for (int i = 0; i < 3; ++i) {
88 for (int j = 0; j < 3; ++j) {
89 small[i][j] = board[sr * 3 + i][sc* 3 + j];
90 }
91 }
92
93 // If player wins this sub-board → record winner
94 if (has_won(small, sym)) {
95 subboard[sr][sc] = sym;
96 }
97 }
98 }
99
100
101 // Check the 3×3 subboard for the final winner
102
103 return has_won(subboard, sym);
104}
