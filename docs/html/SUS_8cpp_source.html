26int SUSBoard::count_new_sus_sequences(int x, int y) {
27 int new_sequences = 0;
28 char current = board[x][y];
29
30 // Only check if the placed letter is 'S' or 'U'
31 if (current != 'S' && current != 'U') {
32 return 0;
33 }
34
35 // Check all possible "S-U-S" patterns that include position (x, y)
36
37 // Horizontal patterns
38 // Pattern: S at x, U at x+1, S at x+2 (current can be any of these)
39 if (y >= 0 && y <= 2) {
40 if (board[x][0] == 'S' && board[x][1] == 'U' && board[x][2] == 'S') {
41 new_sequences++;
42 }
43 }
44
45 // Vertical patterns
46 if (x >= 0 && x <= 2) {
47 if (board[0][y] == 'S' && board[1][y] == 'U' && board[2][y] == 'S') {
48 new_sequences++;
49 }
50 }
51
52 // Main diagonal (top-left to bottom-right)
53 if (x == y) {
54 if (board[0][0] == 'S' && board[1][1] == 'U' && board[2][2] == 'S') {
55 new_sequences++;
56 }
57 }
58
59 // Anti-diagonal (top-right to bottom-left)
60 if (x + y == 2) {
61 if (board[0][2] == 'S' && board[1][1] == 'U' && board[2][0] == 'S') {
62 new_sequences++;
63 }
64 }
65
66 return new_sequences;
67}

69bool SUSBoard::update_board(Move<char>* move) {
70 int x = move->get_x();
71 int y = move->get_y();
72 char mark = move->get_symbol();
73
74 // Validate move coordinates
75 if (x < 0 || x >= rows || y < 0 || y >= columns) {
76 return false;
77 }
78
79 // Check if cell is already occupied
80 if (board[x][y] != blank_symbol) {
81 return false;
82 }
83
84 // Count sequences before placing
85 int total_before = player1_score + player2_score;
86
87 // Apply move
88 n_moves++;
89 board[x][y] = mark;
90
91 // Count new sequences formed
92 int new_sus = count_new_sus_sequences(x, y);
93
94 // Update score for the current player
95 // Mark is either 'S' or 'U', we need to determine which player
96 // Player 1 uses symbol 'X', Player 2 uses symbol 'O'
97 // But in this game, both players can place S or U
98 // We'll track by alternating turns
99 if (n_moves % 2 == 1) {
100 // Player 1's turn (odd moves)
101 player1_score += new_sus;
102 } else {
103 // Player 2's turn (even moves)
104 player2_score += new_sus;
105 }
106
107 return true;
